import React, { useState, useEffect } from 'react';
import { Users, DollarSign, Calendar, AlertCircle, Download, Lock, Shield, LogOut, FileSpreadsheet, MessageCircle, Check, X, Plus, Trash2, Edit2, Eye, Settings } from 'lucide-react';

export default function CDSPhase1() {
  // Auth State
  const [isAuthenticated, setIsAuthenticated] = useState(false);
  const [userRole, setUserRole] = useState(null); // 'president' or 'member'
  const [currentUser, setCurrentUser] = useState(null);
  const [showPresidentSetup, setShowPresidentSetup] = useState(false);
  const [loginForm, setLoginForm] = useState({
    batch: 'A',
    stateCode: '',
    phoneDigits: '',
    password: ''
  });
  const [presidentSetupForm, setPresidentSetupForm] = useState({
    batch: 'A',
    stateCode: '',
    password: '',
    confirmPassword: ''
  });

  // Data State
  const [members, setMembers] = useState([]);
  const [presidentProfile, setPresidentProfile] = useState(null);
  const [currentMonth, setCurrentMonth] = useState(new Date().toISOString().slice(0, 7));
  const [monthlyData, setMonthlyData] = useState({});
  
  // Settings
  const [settings, setSettings] = useState({
    monthlyDues: 1000,
    latenessFine: 500,
    missedMeetingFine: 1000,
    missedSensitizationFine: 2000
  });

  // UI State
  const [activeTab, setActiveTab] = useState('dashboard');
  const [showAddMember, setShowAddMember] = useState(false);
  const [saveStatus, setSaveStatus] = useState('saved');
  const [newMember, setNewMember] = useState({
    name: '',
    phone: '',
    batch: 'A',
    stateCode: '',
    department: '',
    year: ''
  });

  // Generate username: Batch + StateCode + Last4Digits
  const generateUsername = (batch, stateCode, phone) => {
    const last4 = phone.slice(-4);
    return `${batch}${stateCode}${last4}`.toUpperCase();
  };

  // Load data
  useEffect(() => {
    loadData();
  }, []);

  // Auto-save
  useEffect(() => {
    if (isAuthenticated && userRole === 'president' && members.length > 0) {
      saveData('cds_members_v5', members);
    }
  }, [members]);

  useEffect(() => {
    if (presidentProfile) {
      saveData('cds_president_v5', presidentProfile);
    }
  }, [presidentProfile]);

  useEffect(() => {
    if (isAuthenticated && Object.keys(monthlyData).length > 0) {
      saveData('cds_monthly_v5', monthlyData);
    }
  }, [monthlyData]);

  useEffect(() => {
    if (isAuthenticated) {
      saveData('cds_settings_v5', settings);
    }
  }, [settings]);

  const loadData = async () => {
    try {
      const [membersRes, presidentRes, monthlyRes, settingsRes] = await Promise.all([
        window.storage.get('cds_members_v5').catch(() => null),
        window.storage.get('cds_president_v5').catch(() => null),
        window.storage.get('cds_monthly_v5').catch(() => null),
        window.storage.get('cds_settings_v5').catch(() => null)
      ]);

      if (membersRes?.value) setMembers(JSON.parse(membersRes.value));
      if (presidentRes?.value) setPresidentProfile(JSON.parse(presidentRes.value));
      if (monthlyRes?.value) setMonthlyData(JSON.parse(monthlyRes.value));
      if (settingsRes?.value) setSettings(JSON.parse(settingsRes.value));
    } catch (error) {
      console.error('Error loading data:', error);
    }
  };

  const saveData = async (key, data) => {
    setSaveStatus('saving');
    try {
      const result = await window.storage.set(key, JSON.stringify(data));
      if (result) {
        setSaveStatus('saved');
        // Reset to 'saved' after showing briefly
        setTimeout(() => {
          if (saveStatus === 'saved') setSaveStatus('saved');
        }, 2000);
        return true;
      }
      setSaveStatus('error');
      console.error('Save failed: No result returned');
      return false;
    } catch (error) {
      console.error('Error saving:', error);
      setSaveStatus('error');
      // Auto-retry after 3 seconds
      setTimeout(() => {
        setSaveStatus('saved'); // Reset status
      }, 3000);
      return false;
    }
  };

  // President Setup
  const handlePresidentSetup = () => {
    const { batch, stateCode, password, confirmPassword } = presidentSetupForm;
    
    if (!batch || !stateCode || !password) {
      alert('Please fill all fields');
      return;
    }

    if (stateCode.length !== 4) {
      alert('State code must be exactly 4 digits');
      return;
    }

    if (password !== confirmPassword) {
      alert('Passwords do not match');
      return;
    }

    if (password.length < 4) {
      alert('Password must be at least 4 characters');
      return;
    }

    const profile = {
      name: 'President',
      batch,
      stateCode,
      password,
      setupDate: new Date().toISOString()
    };

    setPresidentProfile(profile);
    setShowPresidentSetup(false);
    alert('✅ President profile created!\n\nYou can now login with your credentials.');
  };

  // Authentication
  const handleLogin = () => {
    const { batch, stateCode, phoneDigits, password } = loginForm;
    
    if (!batch || !stateCode || !phoneDigits) {
      alert('Please fill all fields');
      return;
    }

    if (stateCode.length !== 4) {
      alert('State code must be exactly 4 digits');
      return;
    }

    if (phoneDigits.length !== 4) {
      alert('Phone digits must be exactly 4 digits');
      return;
    }

    // Check if president exists, if not show setup
    if (!presidentProfile) {
      alert('President profile not set up. Please complete setup first.');
      setShowPresidentSetup(true);
      return;
    }

    // President login - compare with president profile
    const presidentUsername = `${presidentProfile.batch}${presidentProfile.stateCode}0000`;
    const loginUsername = `${batch}${stateCode}${phoneDigits}`;
    
    if (loginUsername === `${presidentProfile.batch}${presidentProfile.stateCode}0000` && 
        password === presidentProfile.password) {
      setIsAuthenticated(true);
      setUserRole('president');
      setCurrentUser({ 
        name: 'President', 
        batch: presidentProfile.batch,
        stateCode: presidentProfile.stateCode,
        username: presidentUsername
      });
      alert('✅ Welcome, Mr. President!');
      return;
    }
    
    // Member login
    const username = generateUsername(batch, stateCode, phoneDigits);
    const member = members.find(m => m.username === username && m.password === password);
    
    if (member) {
      setIsAuthenticated(true);
      setUserRole('member');
      setCurrentUser(member);
      alert(`✅ Welcome, ${member.name}!`);
      return;
    }
    
    alert('❌ Invalid login credentials!');
  };

  const handleLogout = () => {
    setIsAuthenticated(false);
    setUserRole(null);
    setCurrentUser(null);
    setLoginForm({ batch: 'A', stateCode: '', phoneDigits: '', password: '' });
    setActiveTab('dashboard');
  };

  // Add Member
  const addMember = () => {
    const { name, phone, batch, stateCode, department, year } = newMember;
    
    if (!name || !phone || !batch || !stateCode || !year) {
      alert('Please fill all required fields (name, phone, batch, state code, year)');
      return;
    }

    if (phone.length < 10) {
      alert('Please enter a valid phone number (at least 10 digits)');
      return;
    }

    if (stateCode.length !== 4) {
      alert('State code must be exactly 4 digits');
      return;
    }
    
    const username = generateUsername(batch, stateCode, phone);
    const defaultPassword = phone.slice(-4);
    
    // Check if username already exists
    if (members.some(m => m.username === username)) {
      alert(`Username ${username} already exists! This member may already be registered.`);
      return;
    }
    
    const member = {
      id: Date.now().toString(),
      name,
      phone,
      batch,
      stateCode,
      department: department || 'General',
      year,
      username,
      password: defaultPassword,
      joinedDate: new Date().toISOString(),
      status: 'active'
    };
    
    setMembers([...members, member]);
    setNewMember({ name: '', phone: '', batch: 'A', stateCode: '', department: '', year: '' });
    setShowAddMember(false);
    alert(`✅ Member added successfully!\n\nLogin Details:\nBatch: ${batch}\nState Code: ${stateCode}\nPhone Digits: ${defaultPassword}\nPassword: ${defaultPassword}\n\nUsername: ${username}`);
  };

  // Delete Member
  const deleteMember = (id) => {
    if (confirm('Delete this member? This action cannot be undone.')) {
      setMembers(members.filter(m => m.id !== id));
      alert('✅ Member deleted');
    }
  };

  // Get month data (Week 2, Week 3 meetings + Week 4 sensitization)
  const getMonthData = (monthKey) => {
    if (!monthlyData[monthKey]) {
      return {
        meetings: { week2: {}, week3: {} },
        lateness: { week2: {}, week3: {} },
        sensitization: {},
        exemptions: {}
      };
    }
    return monthlyData[monthKey];
  };

  // Toggle attendance
  const toggleAttendance = (memberId, week, type) => {
    const monthKey = currentMonth;
    const data = getMonthData(monthKey);
    
    if (type === 'meeting') {
      data.meetings[week] = {
        ...data.meetings[week],
        [memberId]: !data.meetings[week][memberId]
      };
    } else if (type === 'lateness') {
      data.lateness[week] = {
        ...data.lateness[week],
        [memberId]: !data.lateness[week][memberId]
      };
    } else if (type === 'sensitization') {
      data.sensitization = {
        ...data.sensitization,
        [memberId]: !data.sensitization[memberId]
      };
    }
    
    setMonthlyData({
      ...monthlyData,
      [monthKey]: data
    });
  };

  // Set exemption
  const setExemption = (memberId, reason) => {
    const monthKey = currentMonth;
    const data = getMonthData(monthKey);
    
    if (reason.trim()) {
      data.exemptions[memberId] = reason;
    } else {
      delete data.exemptions[memberId];
    }
    
    setMonthlyData({
      ...monthlyData,
      [monthKey]: data
    });
  };

  // Calculate monthly owing
  const calculateMonthlyOwing = (member, monthKey) => {
    const data = getMonthData(monthKey);
    const isExempted = !!data.exemptions[member.id];
    
    let total = 0;
    const breakdown = {
      dues: settings.monthlyDues,
      lateness: 0,
      missedMeetings: 0,
      missedSensitization: 0
    };

    total += settings.monthlyDues;

    if (!isExempted) {
      // Count lateness (Week 2 & 3 only)
      ['week2', 'week3'].forEach(week => {
        if (data.lateness[week][member.id]) {
          breakdown.lateness += settings.latenessFine;
          total += settings.latenessFine;
        }
      });

      // Count missed meetings (Week 2 & 3 only)
      ['week2', 'week3'].forEach(week => {
        if (!data.meetings[week][member.id]) {
          breakdown.missedMeetings += settings.missedMeetingFine;
          total += settings.missedMeetingFine;
        }
      });

      // Count missed sensitization (Week 4)
      if (!data.sensitization[member.id]) {
        breakdown.missedSensitization += settings.missedSensitizationFine;
        total += settings.missedSensitizationFine;
      }
    }

    return { total, breakdown, isExempted };
  };

  // Get member attendance
  const getMemberAttendance = (member, monthKey) => {
    const data = getMonthData(monthKey);
    return {
      week2: {
        present: data.meetings.week2[member.id] || false,
        late: data.lateness.week2[member.id] || false
      },
      week3: {
        present: data.meetings.week3[member.id] || false,
        late: data.lateness.week3[member.id] || false
      },
      sensitization: data.sensitization[member.id] || false,
      exemption: data.exemptions[member.id] || null
    };
  };

  // Copy WhatsApp message
  const copyWhatsAppMessage = (member) => {
    const owing = calculateMonthlyOwing(member, currentMonth);
    
    if (owing.total === 0) {
      alert('✅ This member has no outstanding payments!');
      return;
    }

    const monthName = new Date(currentMonth + '-01').toLocaleDateString('en-US', { month: 'long', year: 'numeric' });
    
    let message = `Hello ${member.name},\n\n`;
    message += `This is a reminder from *Charity & Gender CDS, Aba North LGA*.\n\n`;
    message += `Your outstanding payments for *${monthName}*:\n\n`;
    
    if (owing.breakdown.dues > 0) {
      message += `• Monthly Dues: ₦${owing.breakdown.dues.toLocaleString()}\n`;
    }
    if (owing.breakdown.lateness > 0) {
      message += `• Lateness Fines: ₦${owing.breakdown.lateness.toLocaleString()}\n`;
    }
    if (owing.breakdown.missedMeetings > 0) {
      message += `• Missed Meetings: ₦${owing.breakdown.missedMeetings.toLocaleString()}\n`;
    }
    if (owing.breakdown.missedSensitization > 0) {
      message += `• Missed Sensitization: ₦${owing.breakdown.missedSensitization.toLocaleString()}\n`;
    }
    
    message += `\n*Total: ₦${owing.total.toLocaleString()}*\n\n`;
    message += `Please make payment to the treasurer.\n\nThank you.\n_President, Charity & Gender CDS_`;

    navigator.clipboard.writeText(message).then(() => {
      alert(`✅ Message copied!\n\nNow paste in WhatsApp and send to ${member.name}`);
    }).catch(() => {
      alert(message);
    });
  };

  // Export to CSV
  const exportToCSV = () => {
    const monthName = new Date(currentMonth + '-01').toLocaleDateString('en-US', { month: 'long', year: 'numeric' });
    let csv = `CHARITY & GENDER CDS - ABA NORTH LGA\nMonthly Financial Report - ${monthName}\n\n`;
    csv += 'ID,Name,Phone,Batch,State,Username,Dues,Lateness,Missed Meetings,Missed Sens,Exempted,Total,Status\n';
    
    members.forEach((member, index) => {
      const owing = calculateMonthlyOwing(member, currentMonth);
      const status = owing.total > 0 ? 'OWING' : 'CLEAR';
      const exempted = owing.isExempted ? 'YES' : 'NO';
      
      csv += `${index + 1},"${member.name}","${member.phone}","Batch ${member.batch}","${member.state}","${member.username}",`;
      csv += `${owing.breakdown.dues},${owing.breakdown.lateness},${owing.breakdown.missedMeetings},${owing.breakdown.missedSensitization},${exempted},`;
      csv += `${owing.total},"${status}"\n`;
    });

    csv += '\n';
    csv += `Total Members,${members.length}\n`;
    csv += `Members Owing,${members.filter(m => calculateMonthlyOwing(m, currentMonth).total > 0).length}\n`;
    csv += `Total Expected,${settings.monthlyDues * members.length}\n`;
    csv += `Total Outstanding,${members.reduce((sum, m) => sum + calculateMonthlyOwing(m, currentMonth).total, 0)}\n`;

    const blob = new Blob([csv], { type: 'text/csv;charset=utf-8;' });
    const link = document.createElement('a');
    const url = URL.createObjectURL(blob);
    link.setAttribute('href', url);
    link.setAttribute('download', `CDS-Report-${currentMonth}.csv`);
    link.style.visibility = 'hidden';
    document.body.appendChild(link);
    link.click();
    document.body.removeChild(link);
    alert('✅ Report downloaded! Open with Excel.');
  };

  const stats = {
    totalMembers: members.length,
    currentMonthOwing: members.reduce((sum, m) => sum + calculateMonthlyOwing(m, currentMonth).total, 0),
    defaulters: members.filter(m => calculateMonthlyOwing(m, currentMonth).total > 0).length
  };

  // Login Screen
  if (!isAuthenticated) {
    // President Setup Screen
    if (showPresidentSetup || !presidentProfile) {
      return (
        <div className="min-h-screen bg-gradient-to-br from-green-50 to-blue-50 flex items-center justify-center p-4">
          <div className="bg-white rounded-lg shadow-2xl p-8 max-w-md w-full">
            <div className="text-center mb-6">
              <Shield size={64} className="mx-auto mb-4 text-green-600" />
              <h1 className="text-3xl font-bold text-green-800 mb-2">President Setup</h1>
              <p className="text-gray-600">First Time Setup - Create Your Login</p>
            </div>

            <div className="space-y-4">
              <div>
                <label className="block text-sm font-medium mb-2">Select Your Batch *</label>
                <select
                  value={presidentSetupForm.batch}
                  onChange={(e) => setPresidentSetupForm({...presidentSetupForm, batch: e.target.value})}
                  className="w-full px-4 py-3 border rounded-lg focus:ring-2 focus:ring-green-500"
                >
                  <option value="A">Batch A</option>
                  <option value="B">Batch B</option>
                  <option value="C">Batch C</option>
                </select>
              </div>

              <div>
                <label className="block text-sm font-medium mb-2">State Code (4 digits) *</label>
                <input
                  type="text"
                  maxLength="4"
                  value={presidentSetupForm.stateCode}
                  onChange={(e) => setPresidentSetupForm({...presidentSetupForm, stateCode: e.target.value.replace(/\D/g, '')})}
                  placeholder="e.g., 1234"
                  className="w-full px-4 py-3 border rounded-lg focus:ring-2 focus:ring-green-500"
                />
                <p className="text-xs text-gray-500 mt-1">Your unique 4-digit state code</p>
              </div>

              <div>
                <label className="block text-sm font-medium mb-2">Create Password *</label>
                <input
                  type="password"
                  value={presidentSetupForm.password}
                  onChange={(e) => setPresidentSetupForm({...presidentSetupForm, password: e.target.value})}
                  placeholder="Minimum 4 characters"
                  className="w-full px-4 py-3 border rounded-lg focus:ring-2 focus:ring-green-500"
                />
              </div>

              <div>
                <label className="block text-sm font-medium mb-2">Confirm Password *</label>
                <input
                  type="password"
                  value={presidentSetupForm.confirmPassword}
                  onChange={(e) => setPresidentSetupForm({...presidentSetupForm, confirmPassword: e.target.value})}
                  onKeyPress={(e) => e.key === 'Enter' && handlePresidentSetup()}
                  placeholder="Re-enter password"
                  className="w-full px-4 py-3 border rounded-lg focus:ring-2 focus:ring-green-500"
                />
              </div>

              <button
                onClick={handlePresidentSetup}
                className="w-full bg-green-600 text-white py-3 rounded-lg hover:bg-green-700 font-medium"
              >
                Complete Setup
              </button>
            </div>

            <div className="mt-6 p-4 bg-blue-50 rounded-lg text-sm">
              <p className="font-medium text-blue-800 mb-2">📝 Setup Instructions:</p>
              <ul className="text-blue-700 space-y-1">
                <li>• Choose your batch (A, B, or C)</li>
                <li>• Enter any 4-digit code you'll remember</li>
                <li>• Create a secure password</li>
                <li>• You'll login with: Batch + State Code + 0000 + Password</li>
              </ul>
              <p className="text-xs text-blue-600 mt-2">
                Example: If Batch C, State Code 1234, you'll login with C-1234-0000
              </p>
            </div>
          </div>
        </div>
      );
    }

    // Regular Login Screen
    return (
      <div className="min-h-screen bg-gradient-to-br from-green-50 to-blue-50 flex items-center justify-center p-4">
        <div className="bg-white rounded-lg shadow-2xl p-8 max-w-md w-full">
          <div className="text-center mb-6">
            <Shield size={64} className="mx-auto mb-4 text-green-600" />
            <h1 className="text-3xl font-bold text-green-800 mb-2">Charity & Gender CDS</h1>
            <p className="text-gray-600">Aba North LGA - Member Portal</p>
          </div>

          <div className="space-y-4">
            <div>
              <label className="block text-sm font-medium mb-2">Select Batch *</label>
              <select
                value={loginForm.batch}
                onChange={(e) => setLoginForm({...loginForm, batch: e.target.value})}
                className="w-full px-4 py-3 border rounded-lg focus:ring-2 focus:ring-green-500"
              >
                <option value="A">Batch A</option>
                <option value="B">Batch B</option>
                <option value="C">Batch C</option>
              </select>
            </div>

            <div>
              <label className="block text-sm font-medium mb-2">State Code (4 digits) *</label>
              <input
                type="text"
                maxLength="4"
                value={loginForm.stateCode}
                onChange={(e) => setLoginForm({...loginForm, stateCode: e.target.value.replace(/\D/g, '')})}
                placeholder="1234"
                className="w-full px-4 py-3 border rounded-lg focus:ring-2 focus:ring-green-500"
              />
            </div>

            <div>
              <label className="block text-sm font-medium mb-2">Phone Last 4 Digits *</label>
              <input
                type="text"
                maxLength="4"
                value={loginForm.phoneDigits}
                onChange={(e) => setLoginForm({...loginForm, phoneDigits: e.target.value.replace(/\D/g, '')})}
                placeholder="5678"
                className="w-full px-4 py-3 border rounded-lg focus:ring-2 focus:ring-green-500"
              />
              <p className="text-xs text-gray-500 mt-1">Last 4 digits of your phone number</p>
            </div>

            <div>
              <label className="block text-sm font-medium mb-2">Password *</label>
              <input
                type="password"
                value={loginForm.password}
                onChange={(e) => setLoginForm({...loginForm, password: e.target.value})}
                onKeyPress={(e) => e.key === 'Enter' && handleLogin()}
                placeholder="Enter your password"
                className="w-full px-4 py-3 border rounded-lg focus:ring-2 focus:ring-green-500"
              />
            </div>

            <button
              onClick={handleLogin}
              className="w-full bg-green-600 text-white py-3 rounded-lg hover:bg-green-700 font-medium"
            >
              Login
            </button>
          </div>

          <div className="mt-6 p-4 bg-gray-50 rounded-lg text-sm">
            <p className="font-medium mb-2 text-gray-700">Login Instructions:</p>
            <p className="text-gray-600 mb-2">
              <strong>President:</strong> Use 0000 for phone digits
            </p>
            <p className="text-gray-600 mb-2">
              <strong>Members:</strong> Use last 4 digits of your phone
            </p>
            <p className="text-xs text-gray-500 mt-2">
              Example: Batch C + State Code 1234 + Phone 5678 = C12345678
            </p>
          </div>
        </div>
      </div>
    );
  }

  // Member View
  if (userRole === 'member') {
    const owing = calculateMonthlyOwing(currentUser, currentMonth);
    const attendance = getMemberAttendance(currentUser, currentMonth);
    const monthName = new Date(currentMonth + '-01').toLocaleDateString('en-US', { month: 'long', year: 'numeric' });

    return (
      <div className="min-h-screen bg-gradient-to-br from-blue-50 to-green-50 p-4">
        <div className="max-w-4xl mx-auto">
          {/* Header */}
          <div className="bg-white rounded-lg shadow-lg p-6 mb-6">
            <div className="flex items-center justify-between mb-4">
              <div>
                <h1 className="text-2xl font-bold text-gray-800">Welcome, {currentUser.name}!</h1>
                <p className="text-gray-600">Batch {currentUser.batch} • State Code: {currentUser.stateCode}</p>
                <p className="text-sm text-gray-500">Username: {currentUser.username}</p>
              </div>
              <button
                onClick={handleLogout}
                className="flex items-center gap-2 px-4 py-2 bg-red-600 text-white rounded-lg hover:bg-red-700"
              >
                <LogOut size={20} />
                Logout
              </button>
            </div>

            <div className="flex items-center gap-4">
              <label className="font-medium">View Month:</label>
              <input
                type="month"
                value={currentMonth}
                onChange={(e) => setCurrentMonth(e.target.value)}
                className="px-3 py-2 border rounded-lg"
              />
              <span className="text-sm text-gray-600">({monthName})</span>
            </div>
          </div>

          {/* Financial Summary */}
          <div className="bg-white rounded-lg shadow-lg p-6 mb-6">
            <h2 className="text-xl font-bold mb-4">Your Financial Summary</h2>
            
            <div className={`text-center p-6 rounded-lg mb-6 ${
              owing.total > 0 ? 'bg-red-50 border-2 border-red-200' : 'bg-green-50 border-2 border-green-200'
            }`}>
              <p className="text-sm text-gray-600 mb-2">Total Amount Owing</p>
              <p className={`text-5xl font-bold ${owing.total > 0 ? 'text-red-600' : 'text-green-600'}`}>
                ₦{owing.total.toLocaleString()}
              </p>
              {owing.total > 0 && (
                <p className="text-sm text-red-600 mt-2">Please pay to the Treasurer</p>
              )}
              {owing.total === 0 && (
